017    package com.organization.automation.project.spice.mix.aop.guice;
018    
019    import org.junit.Assert;
020    import org.junit.Test;
021    
022    import com.organization.automation.project.spice.mix.aop.guice.LogTestStepModule;
023    import com.organization.automation.project.spice.mix.aop.guice.LogTestStepModuleSpecial;
024    import com.organization.automation.project.spice.mix.aop.guice.TCProxy;
025    import com.organization.automation.project.spice.mix.junit.JUnitExtension;
026    import com.organization.automation.project.spice.mix.junit.test.AbstractTestCase;
027    import com.organization.automation.project.spice.mix.junit.test.AssistedAbstractTestCase;
028    import com.organization.automation.project.spice.mix.junit.test.ITestCase;
029    import com.organization.automation.project.spice.mix.junit.test.TCException;
030    
031    public class GuiceProxyUnitTests extends JUnitExtension {
032    
033            /*
034             * Positive cases.
035             */
036            @Test
037            public void testGuiceProxy_PositiveCase() throws TCException {
038                    ITestCase tc = new AbstractTestCase();
039                    TCProxy proxy = new TCProxy();
040                    proxy.transform(tc).run();
041            }
042            
043            @Test
044            public void testGuiceProxy_PositiveCaseWithLogModule() throws TCException {
045                    ITestCase tc = new AbstractTestCase();
046                    TCProxy proxy = new TCProxy(new LogTestStepModule());
047                    proxy.transform(tc).run();
048            }
049            
050            @Test
051            public void testGuiceProxy_PositiveCaseWithConstructorParameter() throws TCException {
052                    String expected = "100";
053                    String actual   = "0";
054                    
055                    ITestCase tc = new TCProxy(new LogTestStepModuleSpecial()).getTCFactory().create(expected);
056                    actual = ((AssistedAbstractTestCase) tc).getId();
057                    tc.run();
058                    
059                    Assert.assertEquals(expected, actual);
060            }       
061    }
