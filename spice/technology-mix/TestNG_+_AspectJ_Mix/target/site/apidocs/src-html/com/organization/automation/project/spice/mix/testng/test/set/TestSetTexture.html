017    package com.organization.automation.project.spice.mix.testng.test.set;
018    
019    import org.apache.log4j.Logger;
020    
021    import com.organization.automation.project.spice.mix.testng.extension.logging.TAFLoggerFactory;
022    
023    public class TestSetTexture {
024            
025            private static final Logger LOGGER = TAFLoggerFactory.getLogger();
026            
027            protected String description;
028            
029            public void setName(String name) {
030                    this.description = name;
031            }
032    
033            public void setUpSuiteTests(String name) {
034                    LOGGER.info("[SUITE]        ===================================================================");
035                    LOGGER.info("[SUITE]                    Test Suite: " + name);
036                    LOGGER.info("[SUITE]        ===================================================================");
037            }
038            
039            public void setUpAllTests() {
040                    LOGGER.info("[SET]          ===================================================================");
041                    LOGGER.info("[SET]                      Test Set: " + description);
042                    LOGGER.info("[SET]          ===================================================================");
043                    
044                    LOGGER.info("[TEST]_____________________");
045                    LOGGER.info("[TEST][BEFORE] [ALL] start");
046                    LOGGER.info("[TEST]         ...");
047                    LOGGER.info("[TEST][BEFORE] [ALL] end");
048                    LOGGER.info("[TEST]_____________________");
049            }
050            
051            public void setUpEachTest() {
052                    LOGGER.info("[TEST]_____________________");
053                    LOGGER.info("[TEST][BEFORE] [EACH] start");
054                    LOGGER.info("[TEST]         ...");
055                    LOGGER.info("[TEST][BEFORE] [EACH] end");
056                    LOGGER.info("[TEST]_____________________");
057                    
058                    LOGGER.info("[TEST]         *******************************************************************");
059            }
060            
061            public void tearDownEachTest() {
062                    LOGGER.info("[TEST]         *******************************************************************");
063                    
064                    LOGGER.info("[TEST]_____________________");
065                    LOGGER.info("[TEST][AFTER]  [EACH] start");
066                    LOGGER.info("[TEST]         ...");
067                    LOGGER.info("[TEST][AFTER]  [EACH] end");
068                    LOGGER.info("[TEST]_____________________");
069            }
070            
071            public void tearDownAllTests() {
072                    LOGGER.info("[TEST]_____________________");
073                    LOGGER.info("[TEST][AFTER]  [ALL] start");
074                    LOGGER.info("[TEST]         ...");
075                    LOGGER.info("[TEST][AFTER]  [ALL] end");
076                    LOGGER.info("[TEST]_____________________");
077    
078                    LOGGER.info("[SET]          ===================================================================");
079                    LOGGER.info("[SET]                                                          Test Set:  [ DONE ]");
080                    LOGGER.info("[SET]          ===================================================================");
081            }
082            
083            public void tearDownSuiteTests() {
084                    LOGGER.info("[SUITE]        ===================================================================");
085                    LOGGER.info("[SUITE]                                                      Test Suite:  [ DONE ]");
086                    LOGGER.info("[SUITE]        ===================================================================");
087            }
088    }
