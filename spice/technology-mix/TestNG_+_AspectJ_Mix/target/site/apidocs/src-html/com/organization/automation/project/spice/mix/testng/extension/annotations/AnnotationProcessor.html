017    package com.organization.automation.project.spice.mix.testng.extension.annotations;
018    
019    import java.lang.reflect.Method;
020    
021    import org.apache.log4j.Logger;
022    
023    import com.organization.automation.project.spice.mix.testng.extension.logging.TAFLoggerFactory;
024    
025    public class AnnotationProcessor {
026            
027            private static final Logger LOGGER = TAFLoggerFactory.getLogger();
028            
029            private String testName;
030            
031            public void processCustomAnnotations(Method method) {
032                    testName = method.getName();
033                    
034                    History history = method.getAnnotation(History.class);
035                    process(history);                       
036            }
037    
038            private void process(History annotation) {
039                    if ((annotation != null) && annotation.enable()) {
040                            LOGGER.info("[TEST] ***************************************************************************************");
041                            LOGGER.info("[TEST] * Test '" + testName + "'"); 
042                            LOGGER.info("[TETS] * created by '" + annotation.author() + 
043                                                                "' on date '" + annotation.date() + 
044                                                                "' at " + annotation.company());
045                            LOGGER.info("[TEST] ***************************************************************************************");
046                    }
047            }
048    }
