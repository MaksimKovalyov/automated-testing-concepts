017    package com.organization.automation.project.spice.mix.junit.test.suite;
018    
019    import junit.framework.JUnit4TestAdapter;
020    import junit.framework.TestResult;
021    import junit.framework.TestSuite;
022    import org.apache.log4j.Logger;
023    
024    import com.organization.automation.project.spice.mix.junit.extension.logging.TAFLoggerFactory;
025    import com.organization.automation.project.spice.mix.junit.system.TestSystem;
026    import com.organization.automation.project.spice.mix.junit.test.set.TestSetTexture;
027    
028    public class TestSuiteTexture {
029            private static final Logger LOGGER = TAFLoggerFactory.getLogger();
030            public static final TestSuite SUITE = new TestSuite(TestSystem.SUITE_NAME);
031            
032            public static boolean runTestSet(Class<? extends TestSetTexture> c) {
033                    logOnSetStart(c.getName());
034                    
035                    TestResult result = new TestResult();
036                    SUITE.runTest(new JUnit4TestAdapter(c), result);
037    
038                    logOnSetEnd(c.getName(), result.wasSuccessful(), result.runCount());
039            
040                    return result.wasSuccessful();
041            }
042            
043            // log methods
044            protected static void logOnSuiteStart() {
045                    LOGGER.info("[SUITE]       ===========================================================");
046                    LOGGER.info("[SUITE]                      Test suite: " + SUITE.getName());
047                    LOGGER.info("[SUITE]       ===========================================================");
048            }
049            
050            protected static void logOnMainSetsStart() {
051                    //LOGGER.info("[TestSuite]   Test sets start processing...");
052            }
053            
054            protected static void logOnSetStart(String nameTS) {
055                    //LOGGER.info("[TestSuite]   Test set " + nameTS + " starts...");
056            }
057            
058            protected static void logOnSetEnd(String nameTS, boolean result, int count) {
059                    //LOGGER.info("[SUITE]       Test set " + nameTS + " has been finished.");
060                    //LOGGER.info("[TestSuite]   Test set info: ");
061                    //LOGGER.info("[TestSuite]                 result    = " + result);
062                    //LOGGER.info("[TestSuite]                 TCs count = " + count);
063                    //LOGGER.info("[TestSuite]   Test set " + TSName + " - [ " + result + " ]");
064            }
