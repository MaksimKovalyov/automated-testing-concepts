017    package com.organization.automation.project.spice.mix.testng.extension.listener;
018    
019    import org.apache.log4j.Logger;
020    import org.testng.ITestResult;
021    import org.testng.TestListenerAdapter;
022    
023    import com.organization.automation.project.spice.mix.testng.extension.annotations.AnnotationProcessor;
024    import com.organization.automation.project.spice.mix.testng.extension.logging.TAFLoggerFactory;
025    
026    public class AutomationProjectTestListener extends TestListenerAdapter {
027    
028            private static final Logger LOGGER = TAFLoggerFactory.getLogger();
029            
030            private final AnnotationProcessor processor = new AnnotationProcessor();
031                    
032            @Override
033            public void onTestFailure(ITestResult tr) {
034                    LOGGER.info("[TEST]         -------------------------------------------------------------------");
035                    LOGGER.info("[TEST]                                                                    [ " + getResult(tr.getStatus()) + " ]"); 
036    
037                    super.onTestFailure(tr);
038            }
039            
040            @Override
041            public void onTestSuccess(ITestResult tr) {
042                    LOGGER.info("[TEST]         -------------------------------------------------------------------");
043                    LOGGER.info("[TEST]                                                                    [ " + getResult(tr.getStatus()) + " ]"); 
044    
045                    super.onTestSuccess(tr);
046            }
047            
048            @Override
049            public void onTestStart(ITestResult result) {
050                    LOGGER.info("[TEST]         Description: " + result.getName()); 
051    
052                    processor.processCustomAnnotations(result.getMethod().getConstructorOrMethod().getMethod());
053                    
054                    super.onTestStart(result);
055            }
056            
057            private String getResult(int status) {
058                    return (status == 1) ? "PASS" : "FAIL";
059            }
060    }
