017    package com.organization.automation.project.spice.mix.aop.guice;
018    
019    import com.google.inject.AbstractModule;
020    import com.google.inject.Guice;
021    import com.organization.automation.project.spice.mix.testng.test.ITestCase;
022    
023    public class TCProxy {
024    
025            private final AbstractModule logModule; 
026            
027            public TCProxy() {
028                    logModule = new LogTestStepModule();
029            }
030            
031            public TCProxy(AbstractModule module) {
032                    logModule = module;
033            }
034            
035            public ITestCase transform(ITestCase tc) {
036                    return Guice.createInjector(logModule).getInstance(tc.getClass());
037            }
038            
039            public ITCFactory getTCFactory() {
040                    return Guice.createInjector(logModule).getInstance(ITCFactory.class);
041            }
042            
043            public IFeatureTCFactory getFeatureTCFactory() {
044                    return Guice.createInjector(logModule).getInstance(IFeatureTCFactory.class);
045            }
046    }
