17  package com.organization.automation.project.spice.mix.testng.extension.annotations;
18  
19  import java.lang.reflect.Method;
20  
21  import org.apache.log4j.Logger;
22  
23  import com.organization.automation.project.spice.mix.testng.extension.logging.TAFLoggerFactory;
24  
25  public class AnnotationProcessor {
26  	
27  	private static final Logger LOGGER = TAFLoggerFactory.getLogger();
28  	
29  	private String testName;
30  	
31  	public void processCustomAnnotations(Method method) {
32  		testName = method.getName();
33  		
34  		History history = method.getAnnotation(History.class);
35  		process(history);			
36  	}
37  
38  	private void process(History annotation) {
39  		if ((annotation != null) && annotation.enable()) {
40  			LOGGER.info("[TEST] ***************************************************************************************");
41  			LOGGER.info("[TEST] * Test '" + testName + "'"); 
42  			LOGGER.info("[TETS] * created by '" + annotation.author() + 
43  							    "' on date '" + annotation.date() + 
44  							    "' at " + annotation.company());
45  			LOGGER.info("[TEST] ***************************************************************************************");
46  		}
47  	}
48  }
