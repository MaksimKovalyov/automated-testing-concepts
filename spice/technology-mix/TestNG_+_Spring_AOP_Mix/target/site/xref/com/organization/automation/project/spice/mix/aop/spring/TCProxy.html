17  package com.organization.automation.project.spice.mix.aop.spring;
18  
19  import org.aopalliance.aop.Advice;
20  import org.springframework.aop.framework.ProxyFactory;
21  
22  import com.organization.automation.project.spice.mix.testng.test.ITestCase;
23  
24  public class TCProxy {
25  
26  	private ProxyFactory factory;
27  	
28  	private final Advice logAdvice;
29  	
30  	public TCProxy() {	
31  		logAdvice = new LogAroundAdvice();
32  	}
33  
34  	public TCProxy(Advice advice) {	
35  		logAdvice = advice;
36  	}
37  	
38  	public ITestCase transform(ITestCase tc) {
39  		// initialization
40  		factory = new ProxyFactory();
41  		Class<?>[] interfaces = new Class[]{ITestCase.class};
42  		
43  		// set up factory
44  		factory.setTarget(tc);
45  		factory.setInterfaces(interfaces);
46  		factory.addAdvice(logAdvice);
47  		factory.setExposeProxy(true);
48  		
49  		return (ITestCase) factory.getProxy(); 
50  	}
51  }
