017    package com.organization.automation.project.spice.mix.junit.system;
018    
019    import java.util.concurrent.TimeUnit;
020    
021    import org.apache.log4j.Logger;
022    
023    import com.organization.automation.project.spice.mix.junit.extension.logging.TAFLoggerFactory;
024    
025    public class TestSystem {
026    
027            private static final Logger LOGGER = TAFLoggerFactory.getLogger();
028    
029            public static final String SUITE_NAME    = "Test Project Suite";
030            public static final String TEST_SET_NAME = "Feature Addition";
031            public static final String INIT_TEST_SET = "Initialization Tests";
032    
033            // min timeout is 5 seconds
034            public static final long MIN_TIMEOUT = 5000;
035    
036            // log methods
037            public static void logOnError(String message) {
038                    LOGGER.error(message);
039            }
040    
041            public static void logOnInfo(String message) {
042                    LOGGER.info(message);
043            }
044    
045            public static void logOnTCStart(String description) {
046                    LOGGER.info("[TEST]        ***********************************************************");
047                    LOGGER.info("[TEST]        Description: " + description);
048            }
049    
050            public static void logOnTCPass() {
051                    LOGGER.info("[TEST]        ----------------------------------------------------------");
052                    LOGGER.info("[TEST]                                                           [ PASS ]");
053                    LOGGER.info("[TEST]        ***********************************************************");
054            }
055    
056            public static void logOnTCFail() {
057                    LOGGER.info("[TEST]        ----------------------------------------------------------");
058                    LOGGER.info("[TEST]                                                           [ FAIL ]");
059                    LOGGER.info("[TEST]        ***********************************************************");
060            }
061    
062            public void pause(long timeout) {
063                    try {
064                            TimeUnit.SECONDS.sleep(timeout);
065                    } catch (InterruptedException e) {
066                            LOGGER.error(e.toString());
067                    }
068            }
069    }
