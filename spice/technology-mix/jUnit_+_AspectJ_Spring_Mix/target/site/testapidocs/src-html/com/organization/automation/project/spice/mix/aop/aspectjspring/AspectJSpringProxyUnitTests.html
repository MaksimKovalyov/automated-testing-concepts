017    package com.organization.automation.project.spice.mix.aop.aspectjspring;
018    
019    import org.junit.Assert;
020    import org.junit.Test;
021    
022    import com.organization.automation.project.spice.mix.aop.aspectjspring.LogAspect;
023    import com.organization.automation.project.spice.mix.aop.aspectjspring.TCProxy;
024    import com.organization.automation.project.spice.mix.aop.aspectjspring.TCSample;
025    import com.organization.automation.project.spice.mix.junit.JUnitExtension;
026    import com.organization.automation.project.spice.mix.junit.test.AbstractTestCase;
027    import com.organization.automation.project.spice.mix.junit.test.ITestCase;
028    import com.organization.automation.project.spice.mix.junit.test.TCException;
029    
030    public class AspectJSpringProxyUnitTests extends JUnitExtension {
031    
032            /*
033             * Positive cases.
034             */
035            @Test
036            public void testAspectJSpringProxy_PositiveCase() throws TCException {
037                    ITestCase tc = new AbstractTestCase();
038                    TCProxy proxy = new TCProxy();
039                    proxy.transform(tc).run();
040            }
041            
042            @Test
043            public void testAspectJSpringProxy_PositiveCaseWithAspect() throws TCException {
044                    ITestCase tc = new AbstractTestCase();
045                    TCProxy proxy = new TCProxy(new LogAspect());
046                    proxy.transform(tc).run();
047            }
048            
049            @Test
050            public void testAspectJSpringProxy_PositiveCaseWithConstructorParameter() throws TCException {
051                    String expected = "100";
052                    String actual   = "0";
053                    
054                    ITestCase tc = new TCSample(expected);
055                    TCProxy proxy = new TCProxy();
056                    ITestCase tcProxy = proxy.transform(tc);
057                    
058                    actual = ((AbstractTestCase) tcProxy).getId();
059                    tcProxy.run();
060                    
061                    Assert.assertEquals(expected, actual);
062            }
063    }
