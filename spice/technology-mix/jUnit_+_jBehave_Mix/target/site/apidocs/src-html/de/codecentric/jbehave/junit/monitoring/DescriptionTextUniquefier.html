001    /*
002     * This little project is designed to make JBehave stories & scenarios show up
003     * in the JUnit view in Eclipse and potentially other IDEs that support custom
004     * test runners.
005     * 
006     * For details, please visit https://github.com/codecentric/jbehave-junit-runner 
007     */
008    package de.codecentric.jbehave.junit.monitoring;
009    
010    import java.util.HashSet;
011    import java.util.Set;
012    
013    public class DescriptionTextUniquefier {
014    
015            Set<String> strings = new HashSet<String>();
016    
017            public String getUniqueDescription(String junitSafeString) {
018                    while (strings.contains(junitSafeString)) {
019                            junitSafeString += '\u200B'; // zero-width-space
020                    }
021                    strings.add(junitSafeString);
022                    return junitSafeString;
023            }
024    
025    }
