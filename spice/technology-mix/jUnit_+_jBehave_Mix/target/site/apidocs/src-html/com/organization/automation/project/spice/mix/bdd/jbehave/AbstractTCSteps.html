017    package com.organization.automation.project.spice.mix.bdd.jbehave;
018    
019    import org.jbehave.core.annotations.Given;
020    import org.jbehave.core.annotations.Named;
021    import org.jbehave.core.annotations.Then;
022    import org.jbehave.core.annotations.When;
023    import org.junit.Assert;
024    
025    import com.organization.automation.project.spice.mix.junit.test.AbstractTestCase;
026    import com.organization.automation.project.spice.mix.junit.test.TCException;
027    
028    public class AbstractTCSteps {
029            
030            private static final String PASS = "PASS";
031            private static final String FAIL = "FAIL";
032            
033            private AbstractTestCase tc = new AbstractTestCase();
034            
035            /*
036             * Abstract TC steps
037             */
038            @Given("abstract test case with id $ID")
039            public void initializeTestCase(@Named("ID") String id) {
040                    tc = new AbstractTestCase(id);
041            }
042    
043            @When("abstract test case run")
044            public void whenAbstractTestCaseRun() throws TCException {
045                    tc.run();
046            }
047    
048            @Then("result should be [$valid]")
049            public void checkTC(@Named("valid") String result) {
050                    Assert.assertEquals(tc.getResult() ? PASS : FAIL, result);
051            }
052            
053            /*
054             * Abstract Exemplified TC steps
055             */
056            @Given("abstract test case with id <id> and name <tc_name>")
057            public void inputTC(@Named("id") String tcID, @Named("tc_name") String tcName) {
058                    tc = new AbstractTestCase(tcID + ":" + tcName);
059            }
060    
061            @Then("result of exemplified TC should be [<result>]")
062            public void checkExamplifiedTC(@Named("result") String result) {
063                    Assert.assertEquals(tc.getResult() ? PASS : FAIL, result);
064            }
065    }
