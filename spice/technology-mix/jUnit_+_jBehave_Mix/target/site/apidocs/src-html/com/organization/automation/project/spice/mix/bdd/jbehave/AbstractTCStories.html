017    package com.organization.automation.project.spice.mix.bdd.jbehave;
018    
019    import static org.jbehave.core.io.CodeLocations.codeLocationFromClass;
020    import static org.jbehave.core.reporters.Format.CONSOLE;
021    import static org.jbehave.core.reporters.Format.TXT;
022    import static org.jbehave.core.reporters.Format.XML;
023    import static org.jbehave.core.reporters.Format.HTML;
024    
025    import java.util.List;
026    import java.util.Locale;
027    import java.util.Properties;
028    
029    import org.jbehave.core.Embeddable;
030    import org.jbehave.core.configuration.Configuration;
031    import org.jbehave.core.configuration.Keywords;
032    import org.jbehave.core.configuration.MostUsefulConfiguration;
033    import org.jbehave.core.i18n.LocalizedKeywords;
034    import org.jbehave.core.io.CodeLocations;
035    import org.jbehave.core.io.LoadFromClasspath;
036    import org.jbehave.core.io.StoryFinder;
037    import org.jbehave.core.junit.JUnitStories;
038    import org.jbehave.core.parsers.RegexStoryParser;
039    import org.jbehave.core.reporters.CrossReference;
040    import org.jbehave.core.reporters.StoryReporterBuilder;
041    import org.jbehave.core.steps.InjectableStepsFactory;
042    import org.jbehave.core.steps.InstanceStepsFactory;
043    import org.jbehave.core.steps.ParameterConverters;
044    
045    /*
046     * Old style.
047     */
048    public class AbstractTCStories extends JUnitStories {
049    
050            private final CrossReference xref = new CrossReference();
051            
052        public AbstractTCStories() {
053            super();
054            configuredEmbedder()
055                    .embedderControls()
056                            .doGenerateViewAfterStories(false)
057                            .doIgnoreFailureInStories(true)
058                    .doIgnoreFailureInView(true)
059                    .useThreads(1)
060                    .useStoryTimeoutInSecs(60);
061        }
062    
063        @Override
064        public Configuration configuration() {
065            Class<? extends Embeddable> embeddableClass = this.getClass();
066            Properties viewResources = new Properties();
067            viewResources.put("decorateNonHtml", "true");
068            ParameterConverters parameterConverters = new ParameterConverters();
069            Keywords keywords = new LocalizedKeywords(Locale.ENGLISH);
070    
071            return new MostUsefulConfiguration()
072                .useStoryLoader(new LoadFromClasspath(embeddableClass))
073                .useStoryParser(new RegexStoryParser(keywords))
074                .useStoryReporterBuilder(
075                            new StoryReporterBuilder()
076                    //.withReporters(new TAFStoryReporter())
077                    .withCodeLocation(CodeLocations.codeLocationFromClass(embeddableClass))
078                    .withDefaultFormats()
079                    .withViewResources(viewResources)
080                    .withFormats(CONSOLE, TXT, HTML, XML)
081                    .withFailureTrace(true)
082                    .withFailureTraceCompression(true)
083                    .withCrossReference(xref))
084                .useParameterConverters(parameterConverters);
085        }
086    
087        @Override
088        public InjectableStepsFactory stepsFactory() {
089            return new InstanceStepsFactory(configuration(), new AbstractTCSteps());
090        }
091    
092        @Override
093        protected List<String> storyPaths() {
094            return new StoryFinder().findPaths(codeLocationFromClass(this.getClass()), "**/*.story", "**/excluded*.story");
095    
096        }
097    }
