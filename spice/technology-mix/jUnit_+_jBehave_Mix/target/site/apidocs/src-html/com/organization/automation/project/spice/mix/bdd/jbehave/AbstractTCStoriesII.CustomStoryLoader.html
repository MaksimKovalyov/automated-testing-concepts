017    package com.organization.automation.project.spice.mix.bdd.jbehave;
018    
019    import static org.jbehave.core.reporters.Format.TXT;
020    import static org.jbehave.core.reporters.Format.HTML;
021    import static org.jbehave.core.reporters.Format.XML;
022    
023    import java.util.Locale;
024    import java.util.Properties;
025    
026    import org.jbehave.core.annotations.Configure;
027    import org.jbehave.core.annotations.UsingEmbedder;
028    import org.jbehave.core.annotations.UsingPaths;
029    import org.jbehave.core.annotations.UsingSteps;
030    import org.jbehave.core.embedder.Embedder;
031    import org.jbehave.core.embedder.StoryControls;
032    import org.jbehave.core.i18n.LocalizedKeywords;
033    import org.jbehave.core.io.LoadFromClasspath;
034    import org.jbehave.core.parsers.RegexStoryParser;
035    import org.jbehave.core.reporters.CrossReference;
036    import org.jbehave.core.reporters.StoryReporterBuilder;
037    import org.junit.runner.RunWith;
038    
039    import com.organization.automation.project.spice.mix.bdd.jbehave.AbstractTCStoriesII.CustomStoryControls;
040    import com.organization.automation.project.spice.mix.bdd.jbehave.AbstractTCStoriesII.CustomReportBuilder;
041    import com.organization.automation.project.spice.mix.bdd.jbehave.AbstractTCStoriesII.CustomStoryLoader;
042    import com.organization.automation.project.spice.mix.bdd.jbehave.AbstractTCStoriesII.CustomStoryParser;
043    
044    import com.organization.automation.project.spice.mix.bdd.jbehave.extension.reporter.TAFStoryReporter;
045    import com.organization.automation.project.spice.mix.bdd.jbehave.extension.runner.ReportingAnnotatedPathRunner;
046    
047    @RunWith(ReportingAnnotatedPathRunner.class)
048    @Configure(storyControls        = CustomStoryControls.class, 
049                       storyLoader          = CustomStoryLoader.class, 
050                       storyReporterBuilder = CustomReportBuilder.class, 
051                       storyParser          = CustomStoryParser.class)
052    @UsingEmbedder(embedder                 = Embedder.class, 
053                               generateViewAfterStories = false, 
054                               ignoreFailureInStories   = true, 
055                               ignoreFailureInView      = true,
056                               storyTimeoutInSecs       = 68, 
057                               threads                  = 1, 
058                               metaFilters              = "-skip")
059    @UsingSteps(instances = { AbstractTCSteps.class })
060    @UsingPaths(searchIn  =   "resources", 
061                            includes  = { "**/*.story" }, 
062                            excludes  = { "**/examples_table*.story" })
063    public class AbstractTCStoriesII {
064    
065        public static class CustomStoryControls extends StoryControls {
066            public CustomStoryControls() {
067                    super();
068                doDryRun(false);
069                doSkipScenariosAfterFailure(false);
070            }
071        }
072    
073        public static class CustomStoryLoader extends LoadFromClasspath {
074            public CustomStoryLoader() {
075                super(Thread.currentThread().getContextClassLoader());
076            }
077        }
078    
079        public static class CustomReportBuilder extends StoryReporterBuilder {
080            private static final Properties RESOURCES = new Properties();
081            
082            public CustomReportBuilder() {
083                    super();
084                RESOURCES.put("decorateNonHtml", "true");
085                
086                this.withDefaultFormats()
087                    .withFormats(//CONSOLE, 
088                                             TXT, 
089                                             HTML, 
090                                             XML)
091                    .withReporters(new TAFStoryReporter())
092                    .withViewResources(RESOURCES)
093                    .withFailureTrace(true)
094                    .withFailureTraceCompression(true)
095                    .withCrossReference(new CrossReference());
096            }
097        }
098    
099        public static class CustomStoryParser extends RegexStoryParser {
100            public CustomStoryParser() {            
101                    super(new LocalizedKeywords(Locale.ENGLISH));
102            }
103        }    
104    }
